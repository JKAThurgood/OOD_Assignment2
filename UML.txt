'========================
' MUSIC MODEL
'========================

package "Music Model" {

    enum Pitch {
        A
        ASharp
        B
        BFlat
        C
        CSharp
        CFlat
        D
        DSharp
        DFlat
        E
        EFlat
        F
        FSharp
        FFlat
        G
        GSharp
        GFlat
    }

    interface MusicEvent {
        +duration : Double
    }

    class Note {
        -pitch : Pitch
        -octave : Int
        --
        +toFrequency() : Double
    }

    class Rest {
    }

    class Measure {
        -notes : List<MusicEvent>
    }

    class ChannelSettings {
        -source : AudioSource
        -measures : List<Measure>
    }

    class AudioSettings {
        -channels : List<ChannelSettings>
        -sampleRate : Int
        -beatsPerMeasure : Int
        -tempo : Int
    }

    MusicEvent <|.. Note
    MusicEvent <|.. Rest

    Note "1" *-- "1" Pitch
    Measure "1" *-- "1..*" MusicEvent
    ChannelSettings "1" *-- "1..*" Measure
    AudioSettings "1" *-- "1..*" ChannelSettings
}

package "Audio Core" {

    interface AudioSource {
        +generateSample(time : Double, frequency : Double) : Double
    }
}


package "Wave Strategies" {

    class SinWave
    class SquareWave
    class SawWave
    class WhiteNoiseWave
}

AudioSource <|-- SinWave
AudioSource <|-- SquareWave
AudioSource <|-- SawWave
AudioSource <|-- WhiteNoiseWave


'========================
' EFFECTS (DECORATOR)
'========================

package "Effects" {

    abstract class EffectDecorator {
        #source : AudioSource
        +generateSample(time : Double, frequency : Double) : Double
    }

    class VolumeEffect {
        -level : Double
    }

    class ADSEffect {
        -attackEnd : Double
        -decayEnd : Double
        -sustain : Double
    }

    class TanhEffect {
        -drive : Double
    }

    class ClipEffect {
        -threshold : Double
    }

    AudioSource <|-- EffectDecorator
    AudioSource --* EffectDecorator

    EffectDecorator <|-- VolumeEffect
    EffectDecorator <|-- ADSEffect
    EffectDecorator <|-- TanhEffect
    EffectDecorator <|-- ClipEffect
}


'========================
' PIPELINE CONNECTION
'========================

ChannelSettings --> AudioSource


'========================
' APPLICATION LAYER
'========================

package "Application" {

    class SongParser {
        +parse(filename : String) : AudioSettings
    }

    class Synthesizer {
        +render(song : AudioSettings) : AudioBuffer
    }

    class AudioPlayer {
        +play(buffer : AudioBuffer)
    }

    class AudioBuffer
}

SongParser --> AudioSettings
Synthesizer --> AudioSettings
Synthesizer --> AudioBuffer
AudioPlayer --> AudioBuffer
