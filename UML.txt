'========================
' MUSIC MODEL
'========================

package "Music Model" {

    enum Pitch {
        A
        ASharp
        B
        C
        CSharp
        D
        DSharp
        E
        F
        FSharp
        G
        GSharp
    }

    class Note {
        -pitch : Pitch
        -octave : Int
        -duration : Double
        --
        +toFrequency() : Double
    }

    class Measure {
        -notes : List<Note>
    }

    class ChannelSettings {
        -source : AudioSource
        -measures : List<Measure>
    }

    class AudioSettings {
        -channels : List<ChannelSettings>
        -sampleRate : Int
        -beatsPerMeasure : Int
        -tempo : Int
    }

    Note "1" *-- "1" Pitch
    Measure "1" *-- "1..*" Note
    ChannelSettings "1" *-- "1..*" Measure
    AudioSettings "1" *-- "1..*" ChannelSettings
}

package "Audio Core" {

    interface AudioSource {
        +generateSample(time : Double, frequency : Double) : Double
    }
}


package "Wave Strategies" {

    class SinWave
    class SquareWave
    class SawWave
    class WhiteNoiseWave
}

AudioSource <|-- SinWave
AudioSource <|-- SquareWave
AudioSource <|-- SawWave
AudioSource <|-- WhiteNoiseWave

'========================
' EFFECTS (DECORATOR)
'========================

package "Effects" {

    abstract class EffectDecorator {
        #source : AudioSource
        +generateSample(time : Double, frequency : Double) : Double
    }

    class VolumeEffect {
        -level : Double
    }

    class ADSEffect {
        -attackEnd : Double
        -decayEnd : Double
        -sustain : Double
    }

    class TanhEffect {
        -drive : Double
    }

    class ClipEffect {
        -threshold : Double
    }

    AudioSource <|-- EffectDecorator
    AudioSource --* EffectDecorator


    EffectDecorator <|-- VolumeEffect
    EffectDecorator <|-- ADSEffect
    EffectDecorator <|-- TanhEffect
    EffectDecorator <|-- ClipEffect
}

'========================
' PIPELINE CONNECTION
'========================

ChannelSettings --> AudioSource

'========================
' APPLICATION LAYER
'========================

package "Application" {

    class SongParser {
        +parse(filename : String) : AudioSettings
    }

    class Synthesizer {
        +render(song : AudioSettings) : AudioBuffer
    }

    class AudioPlayer {
        +play(buffer : AudioBuffer)
    }

    class AudioBuffer
}

SongParser --> AudioSettings
Synthesizer --> AudioSettings
Synthesizer --> AudioBuffer
AudioPlayer --> AudioBuffer

@enduml
