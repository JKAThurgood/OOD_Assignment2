@startuml

skinparam classAttributeIconSize 0


enum Pitch {
    A
    ASharp
    B
    C
    CSharp
    D
    DSharp
    E
    F
    FSharp
    G
    GSharp
}

class Note {
    -pitch : Pitch
    -octave : Int
    -duration : Double
    --
    +toFrequency() : Double
}

Note "1" *-- "1" Pitch

class Measure {
    -notes : List<Note>
}

Measure "1" *-- "1..*" Note

class ChannelSettings {
    -source : AudioSource
    -measures : List<Measure>
}

ChannelSettings "1" *-- "1..*" Measure

class AudioSettings {
    -channels : List<ChannelSettings>
    -sampleRate : Int
    -beatsPerMeasure : Int
    -tempo : Int
}

AudioSettings "1" *-- "1..*" ChannelSettings


'========================
' STRATEGY PATTERN
'========================

interface AudioSource {
    +generateSample(time : Double, frequency : Double) : Double
}

interface WaveStrategy {
    +generateSample(time : Double, frequency : Double) : Double
}

AudioSource *-- WaveStrategy

class SinWave {
    +generateSample(time : Double, frequency : Double) : Double
}

class SquareWave {
    +generateSample(time : Double, frequency : Double) : Double
}

class SawWave {
    +generateSample(time : Double, frequency : Double) : Double
}

class WhiteNoiseWave {
    +generateSample(time : Double, frequency : Double) : Double
}

WaveStrategy <|.. SinWave
WaveStrategy <|.. SquareWave
WaveStrategy <|.. SawWave
WaveStrategy <|.. WhiteNoiseWave


'========================
' DECORATOR PATTERN
'========================

abstract class EffectDecorator {
    #source : AudioSource
    --
    +generateSample(time : Double, frequency : Double) : Double
}

AudioSource *-- EffectDecorator

class VolumeEffect {
    -level : Double
}

class ADSEffect {
    -attackEnd : Double
    -decayEnd : Double
    -sustain : Double
}

class TanhEffect {
    -drive : Double
}

class ClipEffect {
    -threshold : Double
}

EffectDecorator <|-- VolumeEffect
EffectDecorator <|-- ADSEffect
EffectDecorator <|-- TanhEffect
EffectDecorator <|-- ClipEffect

ChannelSettings "1" *-- "1" AudioSource


'========================
' APPLICATION CLASSES
'========================

class SongParser {
    +parse(filename : String) : AudioSettings
}

class Synthesizer {
    +render(song : AudioSettings) : AudioBuffer
}

class AudioPlayer {
    +play(buffer : AudioBuffer) : void
}

class AudioBuffer

SongParser --> AudioSettings
Synthesizer --> AudioSettings
Synthesizer --> AudioBuffer
AudioPlayer --> AudioBuffer
